      C.a_id, S.s_date

FirstSpend AS (

      a_id, 
      MIN(s_date) AS first_spend_date
  FROM 
      SummaryTable
  GROUP BY 
      a_id

HighestSpend AS (

      a_id, 
      s_date AS highest_spend_date

        a_id,
